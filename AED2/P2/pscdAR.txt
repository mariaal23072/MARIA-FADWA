// Constantes globales
CONSTANTE MAX_ALUMNOS = 50
CONSTANTE MAX_PAREJAS = (MAX_ALUMNOS * (MAX_ALUMNOS - 1)) / 2

// Estructura para los candidatos
ESTRUCTURA Pareja:
    entero alumno1
    entero alumno2
    entero beneficio

// Función para calcular el beneficio de una pareja
// Se asume que solucion es un array de tamaño suficiente (MAX_ALUMNOS) pasado por referencia
FUNCIÓN organizacionClase (N, Amistad[][], Compenetracion[][], solucion[]):
    // Inicialización con arrays de tamaño fijo
    Pareja lista_candidatos[MAX_PAREJAS] 
    booleano emparejado[MAX_ALUMNOS]
    
    PARA i DESDE 0 HASTA N-1:
        emparejado[i] = false
        
    entero beneficio_total = 0
    entero num_parejas = 0

    // Generar todas las posibles parejas y guardarlas en el array estático
    PARA i DESDE 0 HASTA N-1:
        PARA j DESDE i+1 HASTA N-1:
            // Calculo con la funcion del beneficio
            beneficio = (Amistad[i][j] + Amistad[j][i]) * (Compenetracion[i][j] + Compenetracion[j][i])
            
            // Crear pareja en la posición actual y avanzar el contador
            lista_candidatos[num_parejas] = Pareja(i, j, beneficio)
            num_parejas = num_parejas + 1

    // Ordenar parejas por beneficio de mayor a menor (Algoritmo de la Burbuja)
    PARA i DESDE 0 HASTA num_parejas - 2:
        PARA j DESDE 0 HASTA num_parejas - i - 2:
            SI lista_candidatos[j].beneficio < lista_candidatos[j + 1].beneficio:
                // Intercambiar posiciones
                Pareja temporal = lista_candidatos[j]
                lista_candidatos[j] = lista_candidatos[j + 1]
                lista_candidatos[j + 1] = temporal

    entero indice_solucion = 0

    // Emparejar alumnos
    PARA k DESDE 0 HASTA num_parejas - 1:
        Pareja p = lista_candidatos[k]
        
        SI NO emparejado[p.alumno1] Y NO emparejado[p.alumno2]:
            // Emparejar a los alumnos
            emparejado[p.alumno1] = true
            emparejado[p.alumno2] = true
            beneficio_total = beneficio_total + p.beneficio

            // Añadirlos directamente a la solucion usando el índice
            solucion[indice_solucion] = p.alumno1
            indice_solucion = indice_solucion + 1
            solucion[indice_solucion] = p.alumno2
            indice_solucion = indice_solucion + 1

    // EN caso IMPAR
    SI N ES IMPAR:
        PARA i DESDE 0 HASTA N-1:
            SI NO emparejado[i]:
                solucion[indice_solucion] = i
                SALIR DEL BUCLE // Rompe la iteración al encontrar al alumno suelto

    // Ahora solo devuelve el beneficio, porque la 'solucion' se rellenó modificando el parámetro original
    DEVOLVER beneficio_total
